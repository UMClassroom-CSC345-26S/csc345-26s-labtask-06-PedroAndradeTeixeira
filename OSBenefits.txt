tff(os_type,type,os: $tType).
tff(origin_type,type,origin: $tType).
tff(computer_type,type,computer: $tType).

tff(linux_decl,type,linux: os).
tff(windows_decl,type,windows: os).
tff(osx_decl,type,osx: os).

tff(unix_decl,type,unix: origin).
tff(dos_decl,type,dos: origin).

tff(mycomputer_decl,type,mycomputer: computer).

tff(originof_decl,type,originof: os > origin).
tff(runs_decl,type,runs: (computer * os) > $o).
tff(benefits_decl,type,benefits: (computer * origin) > $o).

tff(os_exhaustive,axiom,
    ! [X: os] : (X = linux | X = windows | X = osx)).

tff(os_distinct_1,axiom, linux != windows).
tff(os_distinct_2,axiom, linux != osx).
tff(os_distinct_3,axiom, windows != osx).

tff(origin_exhaustive,axiom,
    ! [Y: origin] : (Y = unix | Y = dos)).

tff(origin_distinct,axiom, unix != dos).

tff(origin_linux,axiom, originof(linux) = unix).
tff(origin_windows,axiom, originof(windows) = dos).
tff(origin_osx,axiom, originof(osx) = unix).

tff(all_computers_run_some_os,axiom,
    ! [C: computer] : ? [S: os] : runs(C,S)).

tff(benefit_rule,axiom,
    ! [C: computer, S: os] : (runs(C,S) => benefits(C,originof(S)))).

tff(mycomputer_runs_osx,axiom,
    runs(mycomputer,osx)).

tff(goal,conjecture,
    benefits(mycomputer,unix)).
